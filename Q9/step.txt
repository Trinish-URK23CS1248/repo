Step 1 – Clone (bash)
git clone https://github.com/<username>/<repo-name>.git
cd <repo-name>

Step 2 – Modify a file (Bash)
echo "Updated content" >> README.md
cat README.md
git diff

Step 3 – Commit & push (Bash)
git add README.md
git commit -m "docs: update README with new content"
git push origin main
git log --oneline -3
